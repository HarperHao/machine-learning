折半插入排序
a=list(input("请输入需要排序的数据:"))
b=[]
for i in a:
   b1=int(i)
   b.append(b1)
b.insert(0,-1)
def hs_sort(arr):
   for i in range(2,len(arr)):
       arr[0]=arr[i]
       low=1
       high=i-1
       while low<=high:
           m=round((low+high)/2)
           if arr[0]<arr[m]:
               high=m-1
           else:
               low=m+1
       j=i-1
       while j>=high+1:
           arr[j+1]=arr[j]
           j-=1
       arr[high+1]=arr[0]
   return arr
arr1=hs_sort(b)
print(arr1[1:])
