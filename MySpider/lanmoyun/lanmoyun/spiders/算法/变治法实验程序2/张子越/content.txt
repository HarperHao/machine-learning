   }
   for (i=10;i>=2;i--)   //i表示当前堆的大小，即等待排序的元素的个数
   {
       a[0]=a[i];        
       a[i]=a[1];
       a[1]=a[0];       //将堆中最小元素与最后一个元素进行交换
       sift(a,1,i-1);    //从根节点处继续调整建堆
   }
}
