def fixdown(a,k,n):
   N=n-1
   while 2*k<=N:
       j=2*k
       if j<N and a[j]<a[j+1]:
           j+=1
       if a[k]<a[j]:
           a[k],a[j]=a[j],a[k]
           k=j
       else:
           break
def heapsort(l):
   n=len(l)-1
   for i in range(n//2,0,-1):
       fixdown(l,i,len(l))
   while n>1:
       l[1],l[n]=l[n],l[1]
       fixdown(l,1,n)
       n=n-1
   return l[1:]
