#include <stdio.h>
#include <stdlib.h>
void sift(int *a,int k,int m)
{
   int i,j,finished;
   i=k;
   j=2*i;
   a[0]=a[k];
   finished=0;
   while((j<=m)&&(!finished))
   {
       if ((j<m)&&(a[j+1]<a[j]))
       j++;
       if (a[0]<=a[j]) finished =1;
       else
       {
           a[i]=a[j];
           i=j;            
           j=2*j;
       }
   }
   a[i]=a[0];
}
void heapsort(int *a)
{
   int i;
   for(i=5;i>=1;i--)
       sift(a,i,10);
   for (i=10;i>=2;i--)
   {
       a[0]=a[i];        
       a[i]=a[1];
       a[1]=a[0];
       sift(a,1,i-1);
   }
}
int main()
{
   int a[10]={0,1,3,5,7,9,8,6,4,2};
   int i=0;
   heapsort(a);
   for(i=1;i<10;i++)
   {
       printf("%d ",a[i]);
   }
system("pause");
   return 0;
}
