import numpy as np

n = 4
c = 5
w = [2,1,3,2]
v = [12,10,20,15]

value = np.zeros((c,c+1))

for i in range(1, n + 1):
   for j in range(1, c + 1):
       value[i][j] = value[i - 1][j]
       if j >= w[i - 1] and value[i][j] < value[i - 1][j - w[i - 1]] + v[i - 1]:
           value[i][j] = value[i - 1][j - w[i - 1]] + v[i - 1]
print(value)
